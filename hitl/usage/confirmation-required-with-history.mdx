This example shows how conversation history is preserved across tool confirmation pauses. When `continue_run()` is called after a confirmation, the agent maintains full context from previous interactions.

<Steps>

  <Step title="Create a Python file">
    ```python confirmation_with_history.py
    from agno.agent import Agent
    from agno.models.openai import OpenAIResponses
    from agno.tools import tool
    from agno.db.sqlite import SqliteDb
    from rich.console import Console
    from rich.prompt import Prompt

    console = Console()

    # Define a tool that requires confirmation
    @tool(requires_confirmation=True)
    def send_notification(recipient: str, message: str, priority: str = "normal") -> str:
        """Send a notification to a user.
        
        Args:
            recipient (str): The recipient's email or username
            message (str): The notification message
            priority (str): Priority level (low, normal, high, urgent)
            
        Returns:
            str: Confirmation of sent notification
        """
        console.print(f"\n📧 Sending {priority} priority notification to {recipient}")
        console.print(f"Message: {message}")
        return f"Notification sent to {recipient} with {priority} priority"

    # Create agent with history enabled
    agent = Agent(
        model=OpenAIResponses(id="gpt-4o"),
        tools=[send_notification],
        add_history_to_context=True,  # Enable history preservation
        num_history_runs=5,  # Keep last 5 conversation runs
        instructions=[
            "You are a helpful notification assistant.",
            "Always address users by their name if they've introduced themselves.",
            "Remember context from previous messages when composing notifications.",
            "When users mention urgency or importance, set appropriate priority levels."
        ],
        markdown=True,
        db=SqliteDb(db_file="tmp/notification_agent.db"),  # Persist history
    )

    console.print("💬 [bold green]Notification Assistant[/bold green]")
    console.print("I can help you send notifications. Type 'exit' to quit.\n")

    # First interaction - user introduces themselves
    console.print("[bold]You:[/bold] Hi, I'm Alice from the DevOps team. We're having some server issues.")
    response = agent.run("Hi, I'm Alice from the DevOps team. We're having some server issues.")
    console.print(f"[bold cyan]Assistant:[/bold cyan] {response.content}\n")

    # Second interaction - provide more context
    console.print("[bold]You:[/bold] The production database is showing high latency since 2pm.")
    response = agent.run("The production database is showing high latency since 2pm.")
    console.print(f"[bold cyan]Assistant:[/bold cyan] {response.content}\n")

    # Third interaction - request that will trigger confirmation
    console.print("[bold]You:[/bold] Please notify the backend team lead about this issue immediately.")
    run_response = agent.run("Please notify the backend team lead about this issue immediately.")

    # Handle confirmation
    if run_response.is_paused:
        console.print("\n⚠️  [bold yellow]Tool confirmation required:[/bold yellow]")
        
        for requirement in run_response.active_requirements:
            if requirement.needs_confirmation:
                tool_call = requirement.tool_execution
                console.print(f"Tool: [bold]{tool_call.tool_name}[/bold]")
                console.print(f"Arguments: {tool_call.tool_args}")
                
                # Get user confirmation
                confirm = Prompt.ask("\nApprove this notification?", choices=["y", "n"], default="y")
                
                if confirm == "y":
                    requirement.confirm()
                    console.print("✅ [green]Notification approved[/green]")
                else:
                    requirement.reject()
                    requirement.confirmation_note = "User declined to send the notification"
                    console.print("❌ [red]Notification rejected[/red]")
        
        # Continue the run - the agent remembers:
        # 1. User is Alice from DevOps
        # 2. The issue is about production database latency since 2pm
        # 3. The context about server issues
        console.print("\n[dim]Continuing execution...[/dim]")
        final_response = agent.continue_run(
            run_id=run_response.run_id,
            requirements=run_response.requirements
        )
        
        # The agent's response will maintain all context
        console.print(f"\n[bold cyan]Assistant:[/bold cyan] {final_response.content}")
        # Expected: "Alice, I've sent an urgent notification to the backend team lead about 
        # the production database latency issue that started at 2pm..."

    # Show conversation history is maintained
    console.print("\n[bold]You:[/bold] Can you also notify the on-call engineer?")
    response = agent.run("Can you also notify the on-call engineer?")
    # The agent still knows about Alice, the database issue, and previous notifications
    console.print(f"[bold cyan]Assistant:[/bold cyan] {response.content}")
    ```
  </Step>
